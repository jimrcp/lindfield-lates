/** * ============================================= * LINDFIELD LATES — Google Apps Script Backend * ============================================= * * This script powers the booking form's capacity tracking. * It stores bookings in a Google Sheet and sends email * notifications to the café owner. * * SETUP INSTRUCTIONS: * ------------------- * 1. Go to https://script.google.com * 2. Click "+ New project" * 3. Delete the default code and paste this entire file * 4. Update the CONFIG section below with your details * 5. Click Run → Run function → "setupSheet" * (This creates the Google Sheet — authorise when prompted) * 6. Click Deploy → New deployment * - Type: "Web app" * - Execute as: "Me" * - Who has access: "Anyone" * 7. Click Deploy, then copy the Web App URL * 8. Paste that URL into the HTML file where it says: * const APPS_SCRIPT_URL = 'YOUR_APPS_SCRIPT_URL_HERE'; * * IMPORTANT: Every time you change this code, you must * Deploy → New deployment (not "Manage deployments") to * get a fresh URL that reflects your changes. * * The Google Sheet will be created in your Google Drive * called "Lindfield Lates Bookings". You can view all * bookings there at any time. * ============================================= */ // ============================================= // CONFIG — update these values // ============================================= const CONFIG = { // Email address to receive booking notifications ownerEmail: 'events@lindfieldcoffeeworks.co.uk', // Maximum guests per Friday evening maxCapacity: 40, // Name of the Google Sheet (created automatically) sheetName: 'Lindfield Lates Bookings', }; // ============================================= // SETUP — run this once to create the spreadsheet // ============================================= function setupSheet() { const ss = SpreadsheetApp.create(CONFIG.sheetName); // --- Bookings tab --- const bookings = ss.getActiveSheet(); bookings.setName('Bookings'); bookings.getRange('A1:H1').setValues([[ 'Timestamp', 'Date', 'Name', 'Email', 'Phone', 'Arrival', 'Guests', 'Requirements' ]]); bookings.getRange('A1:H1').setFontWeight('bold'); bookings.setFrozenRows(1); for (let i = 1; i <= 8; i++) bookings.autoResizeColumn(i); // --- Schedule tab --- const schedule = ss.insertSheet('Schedule'); // Headers schedule.getRange('A1:C1').setValues([['Date', 'Event Name', 'Hide']]); schedule.getRange('A1:C1').setFontWeight('bold'); schedule.setFrozenRows(1); // Format column A as UK date (DD/MM/YYYY) schedule.getRange('A:A').setNumberFormat('dd/mm/yyyy'); // Column C: checkboxes (rows 2–50 to start) schedule.getRange('C2:C50').insertCheckboxes(); // Column widths schedule.setColumnWidth(1, 120); // Date schedule.setColumnWidth(2, 200); // Event Name schedule.setColumnWidth(3, 60); // Hide // Add a helper note so Phoebe knows what to do schedule.getRange('A2').setNote( 'Enter dates in DD/MM/YYYY format (e.g. 03/04/2026).\n' + 'Leave "Event Name" blank for a normal Lindfield Lates.\n' + 'Tick "Hide" to remove a date from the booking form.' ); // Store the spreadsheet ID in script properties for later use PropertiesService.getScriptProperties().setProperty('SHEET_ID', ss.getId()); // Set up a daily trigger to grey out past dates in the Schedule tab setupDailyCleanup(); Logger.log('Sheet created: ' + ss.getUrl()); Logger.log('Sheet ID stored. Setup complete!'); } /** * Creates a daily trigger for sheet maintenance. * Run this once (called automatically by setupSheet). */ function setupDailyCleanup() { // Remove any existing triggers to avoid duplicates const triggers = ScriptApp.getProjectTriggers(); triggers.forEach(t => { if (t.getHandlerFunction() === 'greyOutPastDates' || t.getHandlerFunction() === 'dailyMaintenance') { ScriptApp.deleteTrigger(t); } }); // Run daily between 1–2 AM ScriptApp.newTrigger('dailyMaintenance') .timeBased() .everyDays(1) .atHour(1) .create(); // Also run it right now dailyMaintenance(); } /** * Runs all daily maintenance tasks: * - Pre-populates upcoming Fridays in the Schedule tab * - Greys out past dates */ function dailyMaintenance() { populateScheduleFridays(); greyOutPastDates(); } /** * Pre-populates the Schedule tab with the next 8 Fridays. * Only adds dates that aren't already in the sheet — never * overwrites existing rows (preserves Hide ticks and event names). * Phoebe just opens the sheet and ticks Hide or adds an event name. */ function populateScheduleFridays() { const sheet = getScheduleSheet(); if (!sheet) return; const tz = Session.getScriptTimeZone(); // Generate next 8 Fridays as Date objects (noon to avoid DST issues) const upcomingFridays = []; const d = new Date(); d.setHours(12, 0, 0, 0); const day = d.getDay(); let daysUntilFri = (5 - day + 7) % 7; if (daysUntilFri === 0) daysUntilFri = 7; d.setDate(d.getDate() + daysUntilFri); for (let i = 0; i < 8; i++) { upcomingFridays.push(new Date(d)); d.setDate(d.getDate() + 7); } // Build a set of dates already in the sheet const data = sheet.getDataRange().getValues(); const existingDates = new Set(); for (let i = 1; i < data.length; i++) { const rawDate = data[i][0]; if (!rawDate) continue; if (rawDate instanceof Date) { existingDates.add(Utilities.formatDate(rawDate, tz, 'yyyy-MM-dd')); } else { existingDates.add(String(rawDate).trim()); } } // Append any missing Fridays upcomingFridays.forEach(friday => { const dateStr = Utilities.formatDate(friday, tz, 'yyyy-MM-dd'); if (!existingDates.has(dateStr)) { const newRow = sheet.getLastRow() + 1; sheet.getRange(newRow, 1).setValue(friday); sheet.getRange(newRow, 1).setNumberFormat('dd/mm/yyyy'); sheet.getRange(newRow, 3).insertCheckboxes(); } }); } /** * Grey out rows in the Schedule tab where the date is in the past. * Rows with future/today dates are reset to normal (black text). * Called daily by the time-based trigger. */ function greyOutPastDates() { const sheet = getScheduleSheet(); if (!sheet) return; const data = sheet.getDataRange().getValues(); if (data.length <= 1) return; // header only const today = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd'); const pastColour = '#bbbbbb'; const normalColour = '#000000'; for (let i = 1; i < data.length; i++) { const rawDate = data[i][0]; if (!rawDate) continue; let dateStr; if (rawDate instanceof Date) { dateStr = Utilities.formatDate(rawDate, Session.getScriptTimeZone(), 'yyyy-MM-dd'); } else { dateStr = String(rawDate).trim(); } if (!dateStr || dateStr.length < 10) continue; const row = i + 1; // sheet rows are 1-indexed const range = sheet.getRange(row, 1, 1, 3); // columns A–C if (dateStr < today) { range.setFontColor(pastColour); } else { range.setFontColor(normalColour); } } } // ============================================= // HELPERS // ============================================= /** Get the bookings spreadsheet */ function getSheet() { const id = PropertiesService.getScriptProperties().getProperty('SHEET_ID'); if (!id) throw new Error('Sheet not set up. Run setupSheet() first.'); return SpreadsheetApp.openById(id).getSheetByName('Bookings'); } /** Get the schedule spreadsheet */ function getScheduleSheet() { const id = PropertiesService.getScriptProperties().getProperty('SHEET_ID'); if (!id) throw new Error('Sheet not set up. Run setupSheet() first.'); return SpreadsheetApp.openById(id).getSheetByName('Schedule'); } /** * Read all schedule entries from the Schedule tab. * Returns an array of { date, event, hide } objects. * Skips rows with empty dates. */ function getSchedule() { const sheet = getScheduleSheet(); if (!sheet) return []; const data = sheet.getDataRange().getValues(); const entries = []; for (let i = 1; i < data.length; i++) { // skip header const rawDate = data[i][0]; if (!rawDate) continue; // skip empty rows // Normalise date to YYYY-MM-DD string let dateStr; if (rawDate instanceof Date) { dateStr = Utilities.formatDate(rawDate, Session.getScriptTimeZone(), 'yyyy-MM-dd'); } else { dateStr = String(rawDate).trim(); } if (!dateStr || dateStr.length < 10) continue; entries.push({ date: dateStr, event: data[i][1] ? String(data[i][1]).trim() : '', hide: data[i][2] === true || String(data[i][2]).toLowerCase() === 'true', }); } return entries; } /** Count total guests booked for a given date (YYYY-MM-DD) */ function getBookedCount(dateStr) { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); let total = 0; for (let i = 1; i < data.length; i++) { // skip header // Column B (index 1) = Date, Column G (index 6) = Guests const rowDate = data[i][1]; if (rowDate === dateStr) { total += parseInt(data[i][6], 10) || 0; } } return total; } /** Send email notification to the owner */ function sendNotification(booking) { const subject = 'New Lindfield Lates Booking — ' + booking.date; const body = [ 'New booking for Lindfield Lates!', '', 'Name: ' + booking.name, 'Email: ' + booking.email, 'Phone: ' + (booking.phone || 'Not provided'), 'Date: ' + booking.date, 'Arrival: ' + booking.arrival, 'Guests: ' + booking.guests, 'Requirements: ' + (booking.requirements || 'None'), '', '—', 'View all bookings: https://docs.google.com/spreadsheets/d/' + PropertiesService.getScriptProperties().getProperty('SHEET_ID'), ].join('\n'); MailApp.sendEmail(CONFIG.ownerEmail, subject, body); } /** Send confirmation email to the guest */ function sendConfirmation(booking) { const subject = 'Your Lindfield Lates Booking — ' + booking.date; const body = [ 'Hi ' + booking.name + ',', '', 'Thanks for booking Lindfield Lates!', '', 'Here are your details:', ' Date: ' + booking.date, ' Arrival: ' + booking.arrival, ' Guests: ' + booking.guests, booking.requirements ? ' Notes: ' + booking.requirements : '', '', 'Happy Hour runs from 6–7 pm with reduced prices on cocktails, wine & beer.', '', 'See you Friday!', 'Lindfield Coffee Works', 'https://lindfieldcoffeeworks.co.uk', ].filter(Boolean).join('\n'); MailApp.sendEmail(booking.email, subject, body); } // ============================================= // WEB APP ENDPOINTS // ============================================= /** * GET — returns current capacity for a date * Usage: ?date=2026-03-27 */ function doGet(e) { try { const action = (e.parameter && e.parameter.action) || ''; // --- Schedule endpoint: ?action=schedule --- if (action === 'schedule') { const schedule = getSchedule(); return ContentService.createTextOutput(JSON.stringify({ schedule: schedule, })).setMimeType(ContentService.MimeType.JSON); } // --- Capacity endpoint: ?date=YYYY-MM-DD --- const dateStr = (e.parameter && e.parameter.date) || ''; if (!dateStr) { return ContentService.createTextOutput(JSON.stringify({ error: 'Missing date parameter' })) .setMimeType(ContentService.MimeType.JSON); } const booked = getBookedCount(dateStr); return ContentService.createTextOutput(JSON.stringify({ date: dateStr, booked: booked, remaining: Math.max(0, CONFIG.maxCapacity - booked), maxCapacity: CONFIG.maxCapacity, })).setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput(JSON.stringify({ error: err.message })) .setMimeType(ContentService.MimeType.JSON); } } /** * POST — records a new booking * Body: JSON with { name, email, date, arrival, guests, requirements } */ function doPost(e) { try { const booking = JSON.parse(e.postData.contents); // Validate required fields if (!booking.name || !booking.email || !booking.date || !booking.arrival || !booking.guests) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: 'Missing required fields', })).setMimeType(ContentService.MimeType.JSON); } const guestCount = parseInt(booking.guests, 10) || 1; // Lock to prevent two simultaneous bookings from overbooking const lock = LockService.getScriptLock(); lock.waitLock(10000); // wait up to 10 seconds for the lock let currentBooked = 0; try { // Check capacity (inside lock so no race condition) currentBooked = getBookedCount(booking.date); if (currentBooked + guestCount > CONFIG.maxCapacity) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: 'full', booked: currentBooked, remaining: Math.max(0, CONFIG.maxCapacity - currentBooked), })).setMimeType(ContentService.MimeType.JSON); } // Record the booking const sheet = getSheet(); sheet.appendRow([ new Date().toISOString(), // Timestamp booking.date, // Date booking.name, // Name booking.email, // Email booking.phone || '', // Phone booking.arrival, // Arrival guestCount, // Guests booking.requirements || '', // Requirements ]); } finally { lock.releaseLock(); } // Send emails (outside lock — no need to hold it during email delivery) sendNotification(booking); sendConfirmation(booking); const newBooked = currentBooked + guestCount; return ContentService.createTextOutput(JSON.stringify({ success: true, booked: newBooked, remaining: Math.max(0, CONFIG.maxCapacity - newBooked), })).setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: err.message, })).setMimeType(ContentService.MimeType.JSON); } }
